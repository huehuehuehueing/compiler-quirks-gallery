    switch-table¶

    Details

    Source: control-flow/switch-table.c Compiler: armv7-clang2110 (ARM) Scenario: O2 - Standard Optimization (-O2)

    Source Code¶

    switch-table.c
    /*
 * Switch optimization: jump tables vs. binary search vs. if-else.
 *
 * Compiler chooses implementation based on case distribution:
 * - Dense cases: jump table (O(1))
 * - Sparse cases: binary search (O(log n)) or if-else chain
 */

/* Dense cases - likely becomes a jump table */
int day_of_week(int day)
{
    switch (day) {
        case 0: return 'S';  /* Sunday */
        case 1: return 'M';
        case 2: return 'T';
        case 3: return 'W';
        case 4: return 'T';
        case 5: return 'F';
        case 6: return 'S';  /* Saturday */
        default: return '?';
    }
}

/* Sparse cases - likely if-else or binary search */
int sparse_switch(int x)
{
    switch (x) {
        case 1:     return 10;
        case 100:   return 20;
        case 10000: return 30;
        case 999999: return 40;
        default:    return 0;
    }
}

/* Two cases - becomes simple comparison */
int binary_choice(int x)
{
    switch (x) {
        case 0: return -1;
        case 1: return 1;
        default: return 0;
    }
}

    Explanation¶

    the key insight is that the compiler uses three different strategies for implementing switches based on how the cases are distributed.

    day_of_week(int) - Jump Table Strategy¶

    This function uses a jump table because the cases are dense (0-6 with no gaps):

    mov     r1, r0              # Copy input to r1
mov     r0, #63             # Load default return value ('?')
cmp     r1, #6              # Check if input <= 6
ldrls   r0, .LCPI0_0        # If yes, load jump table address
addls   r0, pc, r0          # Add PC to get absolute address
ldrls   r0, [r0, r1, lsl #2] # Load return value from table
bx      lr                  # Return

    How it works: The jump table at the end contains 7 values (83='S', 77='M', 84='T', 87='W', 70='F', 83='S'). The compiler: 1. Checks if the input is in range (0-6) 2. If yes, uses the input as an index into the table (r1, lsl #2 multiplies by 4 since each entry is 4 bytes) 3. Loads the corresponding return value directly

    This is O(1) - constant time regardless of how many cases exist.

    sparse_switch(int) - Binary Search Strategy¶

    This function uses sequential comparisons because cases are sparse (1, 100, 10000, 999999):

    mov     r1, #15             # Build first threshold (10000)
orr     r1, r1, #9984       # r1 = 10000
cmp     r0, r1              # Compare input to 10000
bgt     .LBB1_4             # If greater, jump to check 999999

cmp     r0, #1              # Check if input == 1
moveq   r0, #10             # If yes, return 10
bxeq    lr                  # Return early

cmp     r0, #100            # Check if input == 100
moveq   r0, #20             # If yes, return 20
bxeq    lr                  # Return early

.LBB1_4:
mov     r1, #1808           # Build 10000 again
orr     r1, r1, #8192       
cmp     r0, r1              # Check if input == 10000
moveq   r0, #30             # If yes, return 30
bxeq    lr                  # Return early

ldr     r1, .LCPI1_0        # Load 999999 from memory
cmp     r0, r1              # Check if input == 999999
moveq   r0, #40             # If yes, return 40
bxeq    lr                  # Return early

b       .LBB1_3             # Jump to default case

    Key pattern: The compiler uses early returns with conditional instructions (moveq/bxeq - "move if equal, branch if equal"). When a case matches, it immediately returns without checking remaining cases. This is more efficient than a jump table for sparse values because a jump table would waste memory on unused entries.

    binary_choice(int) - Arithmetic Trick¶

    This function uses a clever arithmetic transformation instead of explicit comparisons:

    sub     r1, r0, #1         # r1 = input - 1
clz     r1, r1             # Count leading zeros in (input - 1)
lsr     r1, r1, #5         # Shift right by 5 (divide by 32)
cmp     r0, #0             # Check if input == 0
mvneq   r1, #0             # If yes, set r1 to -1 (bitwise NOT of 0)
mov     r0, r1             # Move result to return register
bx      lr                  # Return

    What's happening: For a two-case switch, the compiler avoids branches entirely: - clz (count leading zeros) and lsr convert the input into the return value through bit manipulation - The special case for input=
