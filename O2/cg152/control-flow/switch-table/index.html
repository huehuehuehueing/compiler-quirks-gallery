    switch-table¶

    Details

    Source: control-flow/switch-table.c Compiler: cg152 (x86/x64) Scenario: O2 - Standard Optimization (-O2)

    Source Code¶

    switch-table.c
    /*
 * Switch optimization: jump tables vs. binary search vs. if-else.
 *
 * Compiler chooses implementation based on case distribution:
 * - Dense cases: jump table (O(1))
 * - Sparse cases: binary search (O(log n)) or if-else chain
 */

/* Dense cases - likely becomes a jump table */
int day_of_week(int day)
{
    switch (day) {
        case 0: return 'S';  /* Sunday */
        case 1: return 'M';
        case 2: return 'T';
        case 3: return 'W';
        case 4: return 'T';
        case 5: return 'F';
        case 6: return 'S';  /* Saturday */
        default: return '?';
    }
}

/* Sparse cases - likely if-else or binary search */
int sparse_switch(int x)
{
    switch (x) {
        case 1:     return 10;
        case 100:   return 20;
        case 10000: return 30;
        case 999999: return 40;
        default:    return 0;
    }
}

/* Two cases - becomes simple comparison */
int binary_choice(int x)
{
    switch (x) {
        case 0: return -1;
        case 1: return 1;
        default: return 0;
    }
}

    how the compiler optimizes different switch statements.

    Key Insight: Different Switch Strategies¶

    The compiler uses three different strategies depending on how densely packed the case values are:

    1. day_of_week: Dense cases (0-6) jump table (fastest, O(1) lookup)
    2. sparse_switch: Sparse cases (1, 100, 10000, 999999) binary search tree (moderate, O(log n))
    3. binary_choice: Only two cases simple comparisons (straightforward)

    day_of_week: Jump Table Strategy¶

    mov     eax, 63              # eax = '?' (default return value, ASCII 63)
cmp     edi, 6               # compare input with 6
ja      .L1                  # if input > 6, jump to return (out of range)
mov     edi, edi             # (no-op, register setup)
mov     eax, DWORD PTR CSWTCH.1[0+rdi*4]  # load value from table
.L1:
ret

    How it works: - The function receives the input in edi (by calling convention, the first integer parameter goes in edi) - It first checks if the input is greater than 6 using cmp and ja (jump if above) - If valid (0-6), it uses the input as an index into a lookup table (CSWTCH.1), multiplying by 4 because each entry is 4 bytes (a 32-bit integer) - The table at the bottom contains the ASCII values: 83='S', 77='M', 84='T', 87='W', 70='F' - This is O(1) constant time, regardless of how many cases exist

    sparse_switch: Binary Search Tree Strategy¶

    cmp     edi, 10000           # compare with middle value
je      .L8                  # if equal, jump to case 10000
jg      .L7                  # if greater, check upper half

    The compiler builds a decision tree by comparing against strategic pivot values:

    Lower half (values 10000): 

    mov     eax, 10              # assume case 1 (return 10)
cmp     edi, 1
je      .L5                  # if input == 1, return 10
xor     eax, eax             # eax = 0 (default)
mov     edx, 20              # edx = 20 (case 100 return value)
cmp     edi, 100
cmove   eax, edx             # if input == 100, move 20 into eax
ret

    Upper half (values > 10000): 

    .L7:
xor     eax, eax             # eax = 0 (default)
mov     edx, 40              # edx = 40 (case 999999 return value)
cmp     edi, 999999
cmove   eax, edx             # if input == 999999, move 40 into eax
ret

    Key instruction: cmove (conditional move) - This is a branchless instruction: instead of jumping, it conditionally copies a value - cmove eax, edx means "if the previous comparison was equal, move edx into eax" - This avoids branch misprediction penalties on modern CPUs

    binary_choice: Simple Comparison Chain¶

    mov     eax, -1              # eax = -1 (assume case 0)
test    edi, edi             # check if input == 0
je      .L13                 # if zero, return -1
xor     eax, eax             # eax = 0 (default)
cmp     edi, 1               # check if input == 1
sete    al                   # if equal, set al (low byte) to 1
.L13:
ret
