    fast-math¶

    Details

    Source: arithmetic/fast-math.c Compiler: csparc64g1520 (SPARC64) Scenario: O0 - No Optimization (-O0)

    Source Code¶

    fast-math.c
    /*
 * Fast-math optimizations: -ffast-math / -Ofast effects.
 *
 * These flags allow the compiler to reorder floating-point
 * operations, assuming associativity (which floats don't have).
 * Can change results or break special value handling.
 */

/* With -ffast-math, this may be reordered and give different results */
float sum_array(const float *arr, int n)
{
    float sum = 0.0f;
    for (int i = 0; i < n; i++)
        sum += arr[i];
    return sum;
}

/* NaN check may be optimized away with -ffast-math */
int is_nan(float x)
{
    /* With -ffast-math, compiler assumes no NaN values exist,
     * so this comparison may be optimized to always return 0.
     */
    return x != x;
}

/* Division by variable may become multiplication by reciprocal */
float divide_by_y(float x, float y)
{
    /* With -ffast-math: may become x * (1.0f / y)
     * Less accurate but faster on some architectures.
     */
    return x / y;
}

/* Algebraic simplification (inexact with floats!) */
float algebraic(float a, float b, float c)
{
    /* With -ffast-math: (a + b) + c may be reordered to a + (b + c)
     * These can give different results with floating point!
     */
    return (a + b) + c;
}

    Explanation¶

    the key insight is that this is unoptimized code (-O0), so the compiler generates straightforward, inefficient assembly that closely mirrors the source code without any clever optimizations. Let me walk through each function.

    Overall Pattern: Stack-Heavy Unoptimized Code¶

    At -O0, the compiler prioritizes simplicity over efficiency. Notice how every function starts with save %sp, -<size>, %sp this creates a new stack frame (a reserved area of memory for local variables). The compiler stores function parameters and local variables on the stack rather than keeping them in registers, even when registers would be faster.

    sum_array Function¶

    The core loop pattern:

    .L2:                          // Loop condition check
    ld      [%fp+2039], %g2   // Load loop counter i
    ld      [%fp+2183], %g1   // Load n
    cmp     %g2, %g1          // Compare i < n?
    bl      %icc, .L3         // Branch if less (continue loop)
     nop

.L3:                          // Loop body
    ld      [%fp+2039], %g1   // Load i again
    sra     %g1, 0, %g1       // Sign-extend i (unnecessary at -O0)
    sllx    %g1, 2, %g1       // Multiply i by 4 (shift left 2 bits)
    ldx     [%fp+2175], %g2   // Load array pointer
    add     %g2, %g1, %g1     // Calculate &arr[i]
    ld      [%g1], %f8        // Load arr[i] into floating-point register
    ld      [%fp+2043], %f9   // Load current sum
    fadds   %f9, %f8, %f8     // Add: sum += arr[i]
    st      %f8, [%fp+2043]   // Store sum back to stack
    ld      [%fp+2039], %g1   // Load i
    add     %g1, 1, %g1       // Increment i
    st      %g1, [%fp+2039]   // Store i back to stack
    ba,pt   %xcc, .L2         // Jump back to condition check

    What's happening: The loop loads i from the stack, calculates the array element address (arr[i] requires multiplying i by 4 since floats are 4 bytes), loads the element, adds it to sum, and stores everything back to the stack. Then it increments i and repeats.

    Why this is inefficient: The compiler reloads i and sum from the stack multiple times per iteration. An optimized version would keep these in registers throughout the loop, eliminating the repeated ld (load) and st (store) instructions.

    is_nan Function¶

    st      %f1, [%fp+2175]       // Store parameter x to stack
ld      [%fp+2175], %f9       // Load x back into f9
ld      [%fp+2175], %f8       // Load x again into f8
fcmps   %fcc0, %f9, %f8       // Compare x == x (floating-point comparison)
mov     0, %g1                // Set g1 = 0 (default: not NaN)
movne   %fcc0, 1, %g1         // If not equal, set g1 = 1 (is NaN)
and     %g1, 0xff, %g1        // Mask to 8 bits (unnecessary)
sra     %g1, 0, %g1           // Sign-extend (unnecessary)
mov     %g1, %i0              // Move result to return register

    What's happening: The function compares x with itself using fcmps (floating-point compare single). If they're not equal (movne = move if not equal), it sets the return value to 1 (true, it's NaN). Otherwise, it returns 0.

    Key insight: NaN is the only floating-point value that doesn't equal itself, so x != x is a valid N
