    inline-expansion¶

    Details

    Source: control-flow/inline-expansion.c Compiler: avrg1520 (AVR) Scenario: O0 - No Optimization (-O0)

    Source Code¶

    inline-expansion.c
    /*
 * Function inlining: small functions are expanded at call site.
 *
 * Eliminates call overhead and enables further optimizations
 * by exposing the function body to the caller's context.
 */

/* Tiny function - almost certainly inlined */
static inline int square(int x)
{
    return x * x;
}

/* Using static helps inlining (internal linkage) */
static int cube(int x)
{
    return x * x * x;
}

/* Wrapper that uses inlined functions */
int compute(int x)
{
    /* After inlining, this becomes:
     * return (x * x) + (x * x * x);
     * Which may further simplify to:
     * return x * x * (1 + x);
     */
    return square(x) + cube(x);
}

/* Recursive functions are NOT inlined (infinite expansion) */
int fibonacci(int n)
{
    if (n <= 1) return n;
    return fibonacci(n-1) + fibonacci(n-2);
}

/* Large functions may not be inlined (code size cost) */
int large_function(int x)
{
    int result = 0;
    for (int i = 0; i < 100; i++) {
        result += x * i;
        result ^= (result >> 3);
        result += (result << 5);
    }
    return result;
}

    Explanation¶

    the key insight is that this is unoptimized code (-O0), so functions are NOT being inlined despite the source code's expectations. Let me walk through what's happening.

    The Core Issue: No Inlining at -O0¶

    Even though the source code declares square() as inline and cube() as static, the compiler is generating full function calls for both. At -O0 (no optimization), the compiler ignores inlining hints and generates straightforward, unoptimized code. This is the most important thing to understand about this assembly.

    How AVR Functions Work (Brief Overview)¶

    On AVR microcontrollers, functions follow a calling convention: - Parameters are passed in registers r24/r25 (for 16-bit integers) - Return values come back in r24/r25 - Caller-saved registers (r16-r23) must be preserved by the caller if needed - Frame pointer (r28/r29, also called Y) points to the local stack frame

    Walking Through Each Function¶

    square(int) - The Tiny Function¶

    push r28, r29          # Save frame pointer registers
rcall .                # Adjust stack (3 instructions total)
in r28, __SP_L__       # Load stack pointer into Y (frame pointer)
in r29, __SP_H__
std Y+2, r25           # Store parameter on stack
std Y+1, r24
ldd r18, Y+1           # Load parameter back from stack
ldd r19, Y+2
mov r24, r18           # Copy to r24/r25 (first multiply operand)
mov r25, r19
mov r22, r18           # Copy to r22/r23 (second multiply operand)
mov r23, r19
rcall __mulhi3         # Call multiplication library function
pop r29, r28           # Restore frame pointer
ret

    This demonstrates unoptimized inefficiency: the parameter arrives in r24/r25, gets stored to the stack, then immediately loaded back. An optimized version would skip the stack operations entirely and use the registers directly.

    cube(int) - Computing xÂ³¶

    This function does x * x * x by calling the multiplication routine twice: 1. First __mulhi3 computes x * x, result in r24/r25 2. Then loads the original x again from the stack 3. Calls __mulhi3 again to compute (x*x) * x

    Again, lots of unnecessary stack loads and stores.

    compute(int) - Where Inlining Would Help¶

    rcall square(int)      # Call square(x) - NOT inlined
mov r16, r24           # Save result in r16/r17
mov r17, r25
rcall cube(int)        # Call cube(x) - NOT inlined
add r24, r16           # Add the two results
adc r25, r17           # Add with carry for 16-bit addition

    This makes two full function calls when, with inlining, it could compute x*x + x*x*x directly. The compiler uses r16/r17 to temporarily hold the square() result while calling cube().

    fibonacci(int) - Recursive Function¶

    cpi r24, 2             # Compare n with 2
cpc r25, __zero_reg__  # Compare high byte
brge .L8               # Branch if n >= 2
ldd r24, Y+1           # Base case: return n
ldd r25, Y+2
rjmp .L9
.L8:
sbiw r24, 1            # n-1
rcall fibonacci(int)   # Recursive call
mov r16, r24           # Save result
mov r17, r25
sbiw r24, 2            # n-2
rcall fibonacci(int)   # Second recursive call
add r24, r16           # Add results
adc r25, r17
